      class DataClean(Dataloader):
          def __init__(self, file_path):
              super().__init__(file_path)
              
          def clean_data(self, df):
              # Drop columns containing 'url' or 'count'
              drop_columns = [col for col in df.columns if 'url' in col or 'count' in col] 
              df.drop(columns=drop_columns , inplace=True)
              
              # Drop specific columns
              df = df.drop(['photos','details_website','details_industry','details_type','region','faq',
                   'interviews_experience','career_opportunities_distribution','details_founded',
                   'ratings_rated_ceo','details_revenue','competitors','id'],axis=1)
             
              # Fill missing values with mean
              mean_interview_difficulty = df['interview_difficulty'].mean()
              df['interview_difficulty'].fillna(mean_interview_difficulty, inplace=True)
              
              mean_diversity_inclusion = df['diversity_inclusion_score'].mean()
              df['diversity_inclusion_score'].fillna(mean_diversity_inclusion, inplace=True)
              
              # Split 'details_headquarters' into 'headquarters' and 'headquarter_country'
              df[['headquarters', 'headquarter_country']] = df['details_headquarters'].str.split(', ', expand=True)
              df = df.drop(['details_headquarters'],axis=1)
              
              # Convert timestamp to datetime and calculate recency
              df.timestamp = pd.to_datetime(df.timestamp)
              df['Recency'] = (datetime.now() - df['timestamp']).dt.days
              
              print('Data cleaned succesfully')
              
              
              return df
          
          def apply_rating_funct(self, df):
              df_rating_columns = ['ratings_overall','ratings_career_opportunities',
                                   'ratings_compensation_benefits','ratings_cutlure_values',
                                   'ratings_senior_management','ratings_work_life_balance']
              
              for col in df_rating_columns:
                  df[col + '_new'] = df[col].apply(rating_funct)
                      
          def apply_decision(self, df):
              df_decision_columns = ['ratings_ceo_approval','ratings_business_outlook',
                                     'ratings_recommend_to_friend']
              
              for col in df_decision_columns:
                  df[col + '_new'] = df[col].apply(decision)
                  
          def apply_difficulty(self, df):
              df_difficulty_columns = ['interview_difficulty']
              
              for col in df_difficulty_columns:
                  df[col + '_new'] = df[col].apply(difficulty)
                  
          def apply_size(self, df):
              df_size_columns = ['details_size']
              
              for col in df_size_columns:
                  df[col + '_new'] = df[col].apply(size)
                  
      print('Rating function applied succesfully')

      class Analyse(DataClean):
          def __init__(self, file_path):
              super().__init__(file_path)
              
          def analyse_data(self):
              df = self.load_data()
              
              if df is not None:
                  df_cleaned = self.clean_data(df)
                  
                  # Summary statistics
                  print('Summary statistics:')
                  print(df_cleaned.describe())
                  
                  #Distribution of rating overall
                  sb.histplot(df_cleaned['ratings_overall'], kde=True)
                  plt.title('Distribution of overall rating')
                  plt.show()
                  
                  # Factors affecting ratings overall
                  plt.figure(figsize=(12, 8))
                  correlation_features = ['ratings_overall', 'ratings_career_opportunities', 'ratings_work_life_balance',
                                          'ratings_cutlure_values']
                  correlation_df = df_cleaned[correlation_features].corr()
                  sb.heatmap(correlation_df, annot=True, cmap="YlGnBu")
                  plt.title('Correlation Matrix')
                  plt.show()
                  
                  #Visualize data
                  plt.figure(figsize=(10,6))
                  sb.boxplot(x="company_type",y="ratings_overall", data=df_cleaned)
                  plt.title('Distribution of Ratings Across Products/Services')
                  plt.show()
                  
                  # Average ratings by company type
                  size_ratings = df_cleaned.groupby('company_type')['ratings_business_outlook'].mean()
                  print("Average Ratings by Company type:")
                  print(size_ratings)
                  
                  # Bar chart for ratings 
                  fig = plt.figure(figsize=[18, 15])
                  fig.suptitle('Employee Ratings by Company Type',fontsize=16)
                  
                  x1 = fig.add_subplot(221)
                  sb.barplot(x= df_cleaned.company_type, y= df_cleaned.ratings_work_life_balance.values,
                                         data=df_cleaned) 
                  plt.title('Work-Life Balance')
                  plt.xlabel('Distribution of work-life balance')
                  plt.ylabel('Count')
                  
                  x2 = fig.add_subplot(222)
                  sb.barplot(x= df_cleaned.company_type, y= df_cleaned.ratings_career_opportunities.values,

                  plt.xlabel('Distribution of Career Opportunities')
                  plt.ylabel('Count')
                  
                  x3 = fig.add_subplot(223)

                             data=df_cleaned)
                  plt.title('Compensation Benifits')
                  plt.xlabel('Distribution of Compensation Benifits')
                  plt.ylabel('Count')
                  
                  x4 = fig.add_subplot(224)
                  sb.barplot(x= df_cleaned.company_type, y = df_cleaned.ratings_senior_management.values,
                            data=df_cleaned)
                  plt.title('senior_management')
                  plt.xlabel('Distribution of senior management')
                  plt.ylabel('Count')
                  plt.show()

                  top_n = 10

                  print("Top 10 Companies with Highest Overall Ratings:".format(top_n))
                  print(top_companies.to_string(index=False),"\n\n")

                  company_type_review_count = df_cleaned['company_type'].value_counts()
                  print("Count of each Company Type:")
